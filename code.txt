from flask import Flask, jsonify, request

app = Flask(__name__)

class Evento:
    def __init__(self, id_evento, nome, lotacao_maxima):
        self.id = id_evento
        self.nome = nome
        self.lotacao_maxima = lotacao_maxima
        self.inscritos = 0

    def tem_vaga(self):
        return self.inscritos < self.lotacao_maxima

    def realizar_inscricao(self):
       
        if not self.tem_vaga():
            raise ValueError("Lotação máxima atingida. Evento cheio.")
        self.inscritos += 1

    def to_dict(self):
        return {
            "id": self.id,
            "nome": self.nome,
            "lotacao_maxima": self.lotacao_maxima,
            "inscritos": self.inscritos,
            "vagas_disponiveis": self.lotacao_maxima - self.inscritos
        }

@app.route('/eventos/<int:evento_id>/inscrever', methods=['POST'])
def inscrever_em_evento(evento_id):
   
    evento = banco_de_eventos.get(evento_id)
    if not evento:
        return jsonify({"erro": "Evento não encontrado."}), 404

    try:
        
        evento.realizar_inscricao()
        return jsonify({
            "mensagem": "Inscrição realizada com sucesso!",
            "evento": evento.to_dict()
        }), 200
        
    except ValueError as erro:
       
        return jsonify({"erro": str(erro)}), 400
